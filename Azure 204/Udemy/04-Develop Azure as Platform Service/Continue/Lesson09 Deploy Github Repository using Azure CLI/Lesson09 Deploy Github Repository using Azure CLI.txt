--we set the Github Repository as below
$gitrepo="https://github.com/m93enbah/SampleWeb"

--it will create resource group with specify region
az group create --location centralus --name DefaultResourceGroup-WEU

--it will create app servcie plan with specify resource group and service plan category
az appservice plan create --name newappplan2000 --resource-group DefaultResourceGroup-WEU --sku B1

--it will create web app service with specify resource group and app service
az webapp create --name newapp2000 --resource-group DefaultResourceGroup-WEU --plan newappplan2000 

--it will deploy the web app from github repository to app service as below
az webapp deployment source config --name newapp2000 --resource-group DefaultResourceGroup-WEU --repo-url $gitrepo --branch master --manual-integration
